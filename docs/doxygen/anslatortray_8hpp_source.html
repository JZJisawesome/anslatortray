208  std::string smartWordToPig(const std::string &englishWord)
209  {
210  std::string::size_type wordStartIndex {englishWord.find_first_of(Characters::Letters::ALL)};//after any beginning punctuation
211 
212  std::string::size_type wordEndIndex {englishWord.find_last_of(Characters::APOSTROPHE)};//try to find an ending apostrophe for possesion or a constraction
213  if (wordEndIndex == std::string::npos)
214  wordEndIndex = {englishWord.find_last_of(Characters::Letters::ALL) + 1};//otherwise find the last letter
215 
216 
217  std::string pig {wordToPig(englishWord.substr(wordStartIndex, wordEndIndex - wordStartIndex))};//2nd param is count between start and end
218  std::transform(std::begin(pig), std::end(pig), std::begin(pig), tolower);//make all letters in new word lower for now//fixme why no std::tolower
219  if (std::isupper(englishWord.substr(wordStartIndex, wordEndIndex - wordStartIndex)[0]))//if original word had capital
220  pig[0] = {static_cast<char> (std::toupper(pig[0]))};//new word should have capital
221 
222 
223  //prefix punctuation + pigified word + suffix punctuation
224  std::string result {englishWord.substr(0, wordStartIndex)};
225  result += pig;
226  result += englishWord.substr(wordEndIndex);
227 
228  return result;
229  }
